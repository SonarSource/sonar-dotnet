using System;

using System.Text;

namespace MyLibrary
{
    public class Calc
    {
        public static int Add(int left, int right)
        {
            return left + right;
        }

        public static int Multiply(int left, int right)
        {
            return left * right;
        }

        public static int Divide(int left, int right)
        {
            if (right == 0)
            {
                Console.WriteLine("ERROR: Division by zero!");
            }

            return left / right;
        }
